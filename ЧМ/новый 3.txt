void matrix::calcLU() {
   A = new real * [n];

   for (int i = 0; i < n; i++) {
      A[i] = new real[m];
      for (int j = 0; j < n; j++) {
         //A[i] = new real[m];

         if (j > i) {
            A[i][j] = au[j][i + m - j];
            au[j][i + m - j] = 0.0;
         }
         else {
            A[i][j] = al[i][j + m - i];
            al[i][j + m - i] = 0.0;
         }
         if (i == j) al[i][j] = 1.0;
         A[i][i] = di[i];
         //al[i][i] = 1.0;
         printf("%f   ", A[i][j]);
      }
      printf("\n");
   }

   for (int j = 0; j < n; j++) {
      for (int i = 0; i <= j; i++) {
         au[i][j] = A[i][j];

         for (int k = 0; k < i; k++) {
            au[i][j] -= al[i][k] * au[k][j];
         }

      }

      for (int i = j + 1; i < n; i++) {
         al[i][j] = A[i][j];
         for (int k = 0; k < j; k++) {
            al[i][j] -= al[i][k] * au[k][j];
         }

         al[i][j] /= au[j][j];
      }
   }
}

void matrix::calcY(double** L, double* b, double* y, int n)
{
   for (int i = 0; i < n; i++)
   {
      double sum = 0.0;
      for (int j = 0; j < i; j++)
         sum += L[i][j] * y[j];
      y[i] = (b[i] - sum);
      printf("\n%f", y[i]);
   }
}

void matrix::calcX(double** U, double* y, double* x, int n)
{
   printf("\n");
   for (int i = n - 1; i >= 0; i--)
   {
      double sum = 0.0;
      for (int j = i + 1; j < n; j++)
         sum += U[i][j] * x[j];
      x[i] = (y[i] - sum) / U[i][i];
      printf("\n%f", x[i]);
   }
}

void matrix::createGilbert(double** gl, double** gu, double* D, int n, int m)
{
   for (int i = 0; i < n; i++)
   {
      for (int j = i - m, j1 = 0; j1 < m; j++, j1++)
      {
         if (j < 0) gl[i][j1] = 0;
         else gl[i][j1] = 1.0 / (i + j1 - 1);
      }
      for (int k = i + m, j2 = 0; j2 < m; j2++, k++)
      {
         if (k > m) gu[i][j2] = 0;
         else gl[i][j2] = 1.0 / (i + j2 - 1);
      }
      D[i] = 1.0 / (i + i - 1);
   }
}

int calcLead(std::vector <std::vector <real>> matrix, int N, int i)
{
   double lead = matrix[i][i];
   int leadIndex = i;
   for (int k = i + 1; k < N; k++)
      if (abs(matrix[k][i]) > abs(lead))
      {
         lead = matrix[k][i];
         leadIndex = k;
      }
   return leadIndex;
}
